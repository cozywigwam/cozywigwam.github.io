#+BEGIN_SRC js :cmd "org-babel-node"

const media = {
  xs: (...args) => css`
    @media (min-width: 420px) {
      ${ css(...args) }
    }
  `,
  sm: (...args) => css`
    @media (min-width: 768px) {
      ${ css(...args) }
    }
  `,
  md: (...args) => css`
    @media (min-width: 992px) {
      ${ css(...args) }
    }
  `,
  lg: (...args) => css`
    @media (min-width: 1200px) {
      ${ css(...args) }
    }
  `,
  xl: (...args) => css`
    @media (min-width: 1400px) {
      ${ css(...args) }
    }
  `,
  xxl: (...args) => css`
    @media (min-width: 1900px) {
      ${ css(...args) }
    }
  `
}


td {
  text-align: left;
  font-size: 13px;

  ${media.sm`
    font-size: 14px;
  `}
  ${media.lg`
    font-size: 15px;
  `}
  ${media.xl`
    font-size: 16px;
  `}
  ${media.xxl`
    font-size: 17px;
  `}
}
